##### A. Guide d'utilisation 
```
#####################################################################
# Nettoyage de la console #
###########################
clear;
#####################################################################
# Téléchargement du GitHub #
############################
cd;
rm -rf $HOME/archiso;
git clone https://github.com/dexter74/Archiso.git $HOME/archiso;
#####################################################################
# Lancement des scripts #
#########################
cd $HOME/archiso/V2;
sh package.sh;
sh build.sh;
```
